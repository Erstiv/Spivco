def get_via_mercenary(url):
    # This is the nuclear option. 
    # We prepend 'https://r.jina.ai/' to the URL.
    # Jina handles the headless browsing and returns Markdown.
    jina_url = f"https://r.jina.ai/{url}"
    response = requests.get(jina_url, headers=HEADERS)
    
    if "To verify you are a human" in response.text:
        # Even the Mercenary got stopped (rare)
        raise Exception("Mercenary Blocked")
        
    # Convert Jina's Markdown to HTML for pretty display
    html_content = markdown.markdown(response.text)
    return html_content

def get_from_archive(url):
    api_url = f"http://archive.org/wayback/available?url={url}"
    try:
        data = requests.get(api_url).json()
        if data.get('archived_snapshots', {}).get('closest', {}):
            snapshot_url = data['archived_snapshots']['closest']['url']
            # We use the mercenary to read the ARCHIVE snapshot too (cleaner format)
            return get_via_mercenary(snapshot_url), snapshot_url
    except:
        pass
    return None, None

@app.route('/', methods=['GET', 'POST'])
def index():
    if request.method == 'POST':
        target_url = request.form.get('url')
        content = None
        method = 'mercenary'
        
        try:
            # STRATEGY: GO NUCLEAR IMMEDIATELY
            # For hard targets like Politico, we skip the polite knock
            # and send the Mercenary straight away.
            content = get_via_mercenary(target_url)

        except Exception as e:
            print(f"Mercenary failed ({e}). Checking Archive...")
            content, archive_url = get_from_archive(target_url)
            if content:
                method = 'archive'
                target_url = archive_url
            else:
                return render_template_string(HTML_TEMPLATE, error="Target is locked down tight. Not even the Mercenary could get in.")

        return render_template_string(HTML_TEMPLATE, content=content, source=target_url, method=method)

    return render_template_string(HTML_TEMPLATE)
